<script>
const TARGETS = [
  { id: "internal", label: "社内", desc: "上司・同僚向け", emoji: "🏢" },
  { id: "external", label: "社外", desc: "取引先・関係機関向け", emoji: "🤝" },
];

const TONES = [
  { id: "polite", label: "丁寧・標準", desc: "です・ます調", emoji: "📝" },
  { id: "formal", label: "かしこまった敬語", desc: "謙譲語多め", emoji: "🎩" },
  { id: "soft", label: "柔らかめ丁寧", desc: "親しみのある敬語", emoji: "🌸" },
  { id: "concise", label: "簡潔・端的", desc: "要件重視", emoji: "⚡" },
  { id: "kobe", label: "友達・神戸弁", desc: "50代紳士・神戸弁", emoji: "⚓" },
];

const TONE_INSTRUCTIONS = {
  polite: "「です・ます」調の標準的な丁寧語で書け。",
  formal: "謙譲語・尊敬語を適切に使い、非常にかしこまった文体で書け。",
  soft: "丁寧だが堅すぎず親しみのある敬語表現で書け。",
  concise: "要件を端的に伝え、無駄な前置きを省いた簡潔なビジネス文で書け。",
  kobe: "50代男性の紳士的な神戸弁で書け。親しみはあるが品があり、相手への敬意・礼節を忘れない大人のトーン。「〜しとう」「〜やん」「〜やで」「ほんま」等の神戸弁を自然に使いつつ、失礼・乱暴な表現は排除。大阪弁とは区別し柔らかく上品に。ビジネスメール体裁不要、LINEやメッセージ向けの自然な文体で。",
};

const TARGET_INSTRUCTIONS = {
  internal: "社内の上司や同僚に送るメールとして適切な表現にせよ。",
  external: "社外の取引先や関係機関に送る丁寧で礼儀正しい表現にせよ。",
};

let selectedTarget = "internal";
let selectedTone = "polite";

function renderOptions() {
  const tg = document.getElementById("targetGrid");
  tg.innerHTML = TARGETS.map(t =>
    `<div class="option-btn center ${t.id === selectedTarget ? 'active' : ''}" data-target="${t.id}">
      <span class="emoji">${t.emoji}</span>
      <div class="label">${t.label}</div>
      <div class="desc">${t.desc}</div>
    </div>`
  ).join("");

  tg.querySelectorAll(".option-btn").forEach(btn => {
    btn.addEventListener("click", () => { selectedTarget = btn.dataset.target; renderOptions(); });
  });

  const tn = document.getElementById("toneGrid");
  tn.innerHTML = TONES.map((t, i) =>
    `<div class="option-btn ${t.id === selectedTone ? 'active' : ''} ${TONES.length % 2 === 1 && i === TONES.length - 1 ? 'full-width' : ''}" data-tone="${t.id}">
      <span class="emoji">${t.emoji}</span>
      <div>
        <div class="label">${t.label}</div>
        <div class="desc">${t.desc}</div>
      </div>
    </div>`
  ).join("");

  tn.querySelectorAll(".option-btn").forEach(btn => {
    btn.addEventListener("click", () => { selectedTone = btn.dataset.tone; renderOptions(); });
  });
}

function buildPrompt(input) {
  return `カジュアルな文章をビジネスメールに再構成せよ。敬語変換ではなく、要点・目的・依頼事項を読み取り、相手が理解しやすい構成に再構築すること。

トーン:${TONE_INSTRUCTIONS[selectedTone]}
宛先:${TARGET_INSTRUCTIONS[selectedTarget]}
構成:件名(【件名】で開始)→挨拶→結論先行で本文→結び。複数要件は項目整理。依頼は何を/いつまでに/どうするか明示。情報の追加・欠落不可。見出し記号(■●▶等)禁止。大見出し:１．中見出し:(１)小見出し:①。結果のみ出力。

${input}`;
}

const inputEl = document.getElementById("inputText");
const convertBtn = document.getElementById("convertBtn");
const errorBox = document.getElementById("errorBox");
const resultBox = document.getElementById("resultBox");
const resultBody = document.getElementById("resultBody");
const resultTag = document.getElementById("resultTag");
const copyBtn = document.getElementById("copyBtn");

const apiKeyInput = document.getElementById("apiKeyInput");
const toggleKey = document.getElementById("toggleKey");

toggleKey.addEventListener("click", () => {
  if (apiKeyInput.type === "password") {
    apiKeyInput.type = "text";
    toggleKey.textContent = "🙈";
  } else {
    apiKeyInput.type = "password";
    toggleKey.textContent = "👁";
  }
});

inputEl.addEventListener("input", () => {
  convertBtn.disabled = !inputEl.value.trim();
});

convertBtn.addEventListener("click", async () => {
  const input = inputEl.value.trim();
  const apiKey = apiKeyInput.value.trim();
  if (!input) return;
  if (!apiKey) {
    errorBox.textContent = "APIキーを入力してください。";
    errorBox.style.display = "block";
    return;
  }

  convertBtn.disabled = true;
  convertBtn.innerHTML = '⏳ 変換中<span class="loading-dots"></span>';
  errorBox.style.display = "none";
  resultBox.style.display = "none";

  try {
    const resp = await fetch("https://api.anthropic.com/v1/messages", {
      method: "POST",
      headers: { "Content-Type": "application/json", "x-api-key": apiKey, "anthropic-version": "2023-06-01", "anthropic-dangerous-direct-browser-access": "true" },
      body: JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1000,
        messages: [{ role: "user", content: buildPrompt(input) }],
      }),
    });

    const data = await resp.json();
    if (data.content && data.content.length > 0) {
      const text = data.content.filter(i => i.type === "text").map(i => i.text).join("\n");
      const toneLabel = TONES.find(t => t.id === selectedTone)?.label || "";
      const targetLabel = TARGETS.find(t => t.id === selectedTarget)?.label || "";
      resultTag.textContent = `${targetLabel} / ${toneLabel}`;
      resultBody.textContent = text;
      resultBox.style.display = "block";
    } else {
      errorBox.textContent = "変換に失敗しました。もう一度お試しください。";
      errorBox.style.display = "block";
    }
  } catch (err) {
    errorBox.textContent = "エラーが発生しました: " + err.message;
    errorBox.style.display = "block";
  } finally {
    convertBtn.disabled = !inputEl.value.trim();
    convertBtn.innerHTML = "🔄 ビジネスメールに変換";
  }
});

copyBtn.addEventListener("click", () => {
  const text = resultBody.textContent;
  if (navigator.clipboard) {
    navigator.clipboard.writeText(text).then(() => showCopied()).catch(() => fallbackCopy(text));
  } else {
    fallbackCopy(text);
  }
});

function fallbackCopy(text) {
  const ta = document.createElement("textarea");
  ta.value = text; document.body.appendChild(ta); ta.select();
  document.execCommand("copy"); document.body.removeChild(ta); showCopied();
}

function showCopied() {
  copyBtn.textContent = "✓ コピー済";
  copyBtn.classList.add("copied");
  setTimeout(() => { copyBtn.textContent = "コピー"; copyBtn.classList.remove("copied"); }, 2000);
}

renderOptions();
</script>
